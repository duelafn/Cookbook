
\begin{recipe}{Vinaigrette}\tag{salad}
  \rating    3
  \nutrition 1
  \health    2
  \workload{light}
  \time{5}
  \yield{}
  \servings{}
  \source{Alton Brown}
  % \photo{}
  \maketitle

  \begin{ingredients2}
    1 \t & dijon mustard\\
    2 \T & vinegar\\
    & salt\\
    & pepper\\
    & crushed garlic\\
    6 \T & oil
  \end{ingredients2}

  Combine everything but the oil, then add the oil and stir/shake
  vigorously.

\end{recipe}
