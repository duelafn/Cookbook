
\begin{recipe}{Pumpkin Pie}\tag{pie}
  \rating    4
%   \nutrition{}
%   \health{}
%   \workload{}
  \time{75}
  \yield{1 pie}
  \servings{8}
%   \source{}
  \photo{pumpkin-pie}
  \maketitle

  \begin{ingredients2}
    2 & eggs\\
    2--3 \cups & pumpkin\\
    1\half \cups & evaporated milk\\
    \third \cup & sugar\\
    \third \cup & brown sugar\\
    1 \t & cinnamon\\
    \threefourth \t & ginger\\
    \half \t & nutmeg\\
    \fourth \t & clove or allspice\\
    \half \t & salt
  \end{ingredients2}

  If using fresh pumpkin, puree pumpkin with evaporated milk.

  Mix all ingredients. Bake 15 minutes at \tempF{425} then reduce heat and
  cook 40--50 minutes at \tempF[$\le 200$]{350} until knife comes out clean.
\end{recipe}
