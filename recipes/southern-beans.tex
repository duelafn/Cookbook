
\begin{recipe}{Southern Beans}\tag{Sylvia's Beans}
  \rating    4
  \nutrition 3
  \health    3
  \workload{medium}
  \time{60}
  \yield{}
  \servings{}
  \source{Dean Serenevy}
  \maketitle

  \begin{ingredients2}
    \half--1 \lb & bacon\\
    3 cans & black-eyed peas\\
    2 \c & Dashi (page~\pageref{dashi})\\
    1 \t & black pepper\\
    \quarter \t & crushed red pepper\\
    2 & bay leaves
  \end{ingredients2}

  Bring to boil, simmer 1 hour.

  Additions under consideration: 1~\t thyme, garlic, some sugar

% https://www.tasteofhome.com/recipes/southern-black-eyed-peas/
% https://www.southyourmouth.com/2013/09/southern-style-black-eyed-peas.html
\end{recipe}
