
\begin{recipe}{Pie Crust (flour)}%\tag{}
  \rating    4
  \nutrition 0
  \health    1
  \workload{medium}
  \time{45}
  \yield{2 crusts}
  \servings{}
  \source{}
  % \photo{}
  \maketitle

  \begin{ingredients2}
    12 \T & butter\\
    2 \cups & flour\\
    \half \t & salt\\
    4--5 \T & cold water
  \end{ingredients2}

  Mush butter, flour, and salt in bowh with hands until butter completely
  incorporated into flour (no lumps). Stir water into crus 1~\T at a time
  until dough forms. Roll out.
\end{recipe}
