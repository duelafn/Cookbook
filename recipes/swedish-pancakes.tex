
\begin{recipe}{Swedish Pancakes}\tag{pancakes}
  \rating    4
  \nutrition 2
  \health    2
  \workload{medium}
  \time{45}
  \yield{}
  \servings{}
  \source{Schermer}
  % \photo{}
  \maketitle

  \begin{ingredients2}
    4           & eggs\\
    \half \cup  & milk\\
    1 \cup      & flour\\
    1\third \T  & sugar\\
    \half \t    & salt\\
    lingonberry or cranberry sauce
  \end{ingredients2}

  Beat all ingredients except sauce together. Fry cakes in butter, flipping
  once. Serve with lingonberry sauce or whole berry cranberry sauce.
\end{recipe}
