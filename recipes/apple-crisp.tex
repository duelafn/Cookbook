
\begin{recipe}{Apple Crisp}\tag{crisp}\tag{apples}
  \rating    5
  \nutrition 3
  \health    3
  \workload{light}
  \time{30}
  \yield{1 large cake pan}
  \servings{6}
  \source{Dean Serenevy}
  \photo{apple-crisp}
  \maketitle

  \begin{ingredients2}
    10--14       & apples (4--5 \lb)\\
    1\third \c   & oats\\
    \twothird \c & flour\\
    \twothird \c & brown sugar\\
    \third \c    & butter\\
    2 \t         & cinnamon\\
    \half \t     & vanilla powder (opt)\\
    \half \t     & nutmeg
  \end{ingredients2}

  Cut up apples and put in cake pan. Chop butter into the rest of the
  ingredients with your hands and sprinkle over the apples. Bake for 30
  minutes at 350\degF. Top with french vanilla ice cream!
\end{recipe}
