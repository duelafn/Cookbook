
\begin{recipe}{Corn Fritters}%\tag{}
  \rating    4
  \nutrition 2
  \health    2
  \workload{heavy}
  \time{60}
  \yield{8}
  \servings{8}
  \source{Fanny Farmer}
  % \photo{}
  \maketitle

  \begin{ingredients2}
    2 \cups    & corn kernels\\
    2          & eggs\\
    \half \cup & flour\\
    1 \t       & salt\\
    \fourth \t & pepper\\
    1 quart    & canola oil
  \end{ingredients2}

  Heat canola oil to 380\degF in big pot. Mix corn (fresh/frozen/canned),
  eggs, flour, salt, and pepper. By \fourth~cups, deep fry in oil until
  golden. Makes 8 small fritters.
\end{recipe}
