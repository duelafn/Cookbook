
\begin{recipe}{Crab cakes}\tag{seafood}
  \rating    3
  \nutrition 3
  \health    2
  \workload{heavy}
  \time{60}
  \yield{6 cakes}
  \servings{2--3}
  \source{Alton Brown}
  % \photo{}
  \maketitle

  \begin{ingredients2}
    1 \c & ``lump'' crabmeat\\
    1 \c & ``special'' crabmeat\\
    \half \c & mayonnaise\\
    \threefourth \c & panko bread crumbs\\
    \half & lemon\\
    & black pepper\\
    \threefourth \c & panko bread crumbs
  \end{ingredients2}

  Combine ingredients, reserving \threefourth~\c panko for rolling. Form cakes and
  coat in panko. Fry at 375\degF until golden brown (2--3~minutes per side).

  Note: cakes are a bit crumbly.
\end{recipe}
