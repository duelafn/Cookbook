
\begin{recipe}{Chocolate Oatmeal No-Bake Cookies}\tag{cookies}
  % \tag{}
  \rating    3
  \nutrition 3
  \health    2
  \workload{medium}
  \time{30}
  \source{internet}
  % \photo{}
  \maketitle

  \begin{ingredients2}
    2 \cups & sugar\\
    4 \T & cocoa powder\\
    \half \cup & butter\\
    \half \cup & milk\\
    1 \cup & peanut butter\\
    1 \T & vanilla\\
    3 \cups & oatmeal
  \end{ingredients2}

  Boil butter, sugar, cocoa then add rest. Drop onto wax paper and cool.
\end{recipe}
