
\begin{recipe}{Carrot Cake}%\tag{}
  \rating    2
  \nutrition 2
  \health    2
  \workload{medium}
  \time{}
  \yield{}
  \servings{}
  \source{}
  % \photo{}
  \maketitle

  \begin{ingredients2}
    340 \g & flour\\
    340 \g & carrots\\
    1 \t & baking powder\\
    1 \t & baking soda\\
    \fourth \t & allspice\\
    \half \t & cinnamon\\
    \fourth \t & nutmeg\\
    \half \t & salt\\
    285 \g & sugar\\
    60 \g & brown sugar\\
    3 & eggs\\
    170 \g & yogurt\\
    170 \g & vegetable oil\\
    225 \g & cream cheese\\
    4 \T & butter\\
    1 \T & vanilla\\
    250 \g & powdered sugar
  \end{ingredients2}

  Butter and flour a 9\inch round pan.

  Combine flower, powder, soda, and spices then toss with grated carrots.

  Add sugar, eggs, yogurt, oil and stir just enough to combine.

  Bake 45~minutes at 350\degF then 20~minutes at \tempF[205]{325}.

  Cool cake 15 minutes before turning out, then cool completely before frosting.

  Mix remaining ingredients to make frosting.
\end{recipe}
