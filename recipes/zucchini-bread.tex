
\begin{recipe}{Zucchini Bread}%\tag{}
  \rating    0
  \nutrition 4
  \health    3
  \workload{medium}
  \time{90}
  \yield{1 loaf}
  \servings{4}
  \source{\Joy}
  \photo{zucchini-bread}
  \maketitle

  \begin{ingredients2}
    \fourth \t & salt\\
    2 \c & grated zucchini\\
    \threefourth \c & sugar\\
    2 & eggs\\
    \half \c & vegetable oil\\
    1 \T & vanilla\\
    1\fourth \c & flour\\
    1 \t & baking soda\\
    1 \t & baking powder\\
    1 \t & cinnamon\\
    \fourth \t & nutmeg
  \end{ingredients2}

  Grate zucchini, seeds (even large ones) are fine. Add salt and let drain in
  cheesecloth in a strainer.

  Beat sugar, eggs, oil, and vanilla until somewhat airy. Stir in dry ingredients.
  Squeeze zucchini in cheesecloth to drain as much liquid as possible. Stir in dry
  zucchini.

  Bake in greased and floured loaf pan about 45--60~minutes \tempF[190]{350} until
  bread pulls away from the pan.
\end{recipe}
