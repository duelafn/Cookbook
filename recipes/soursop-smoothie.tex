
\begin{recipe}{Soursop Smoothie}\tag{guanabana smoothie}
  \rating    4
  \nutrition 0
  \health    0
  \workload{light}
  \time{20}
  \yield{8~cups}
  \servings{4}
  \source{Tropical Fruit Box}
  \maketitle
  % \photo{}

  \begin{ingredients2}
    1 & ripe soursop\\
    5 \c & water\\
    1 can & sweetened condensed milk\\
    2 \T & lime juice\\
    1 \T & vanilla\\
    1 \t & nutmeg
  \end{ingredients2}

  Peel and remove seeds from soursop. Puree fruit in blender with 3~cups water. Pour
  into bowl and stir in remaining ingredients, garnish with nutmeg.
\end{recipe}
