
\begin{recipe}{Garlic Bread}\tag{bread}
  \rating    5
  \nutrition
  \health
  \workload{medium}
  \time{30}
  \yield{}
  \servings{}
  \source{}
  \maketitle
  % \photo{}

  \begin{ingredients2}
    ~ & pizza crust or bread\\
    2\half \T & butter\\
    6--8 cloves & garlic
  \end{ingredients2}

  Cook the garlic in the butter. Spread over bread and sprinkle with
  cheese.
\end{recipe}
