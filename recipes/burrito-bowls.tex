
\begin{recipe}{Burrito Bowls}%\tag{}
  \rating    4
  \nutrition 3
  \health    3
  \workload{medium}
  \time{45}
  \yield{}
  \servings{4}
  \source{Hello Fresh}
  \maketitle
  % \photo{}

  \begin{ingredients2}
    1 \cup & rice\\
    2 & poblano peppers\\
    2 & tomatoes\\
    20 \oz & meat\\
    1 \T & chipotle\\
    1 \T & chili powder\\
    & cilantro\\
    1 & lemon\\
    & sour cream
  \end{ingredients2}

  \begin{itemize}[nosep]
  \item Cook rice in 1\half~\cups water.
  \item Chop tomatoes and combine with some lemon juice.
  \item Fry poblano until softened and lightly charred (7--9 min). Transfer
    to plate.
  \item Cook meat with chipotle, chili powder, and some water.
  \item Fluff rice with some lemon zest and chopped cilantro. Top rice with
    stuff.
  \end{itemize}
\end{recipe}
