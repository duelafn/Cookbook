
\begin{recipe}{Scones}\tag{blueberry}\tag{muffins}
  \rating    4
  \nutrition 3
  \health    2
  \workload{medium}
  \time{45}
  \source{Dean Serenevy}
  % \photo{}
  \maketitle

  \begin{ingredients2}
    1 stick     & butter\\
    2 \cups     & flour\\
    2 \T        & sugar\\
    1 \T        & baking powder\\
    1 \t        & salt\\
                & cinnamon\\
    2           & eggs\\
                & vanilla\\
    \third \cup & milk or cream\\
                & blueberries\\
    4 \oz       & cream cheese (opt)
  \end{ingredients2}

  ``Cut'' butter into dry ingredients (with hands until butter fully
  incorporated -- no butter lumps). Stir in wet ingredients adding just
  enough milk to form a paste. Stir in blueberries and then, optionally,
  cream cheese.

  Spoon in to buttered and floured muffin pan. Let rest while you make the
  \reciperef{Oat-Brown Sugar Topping}. Sprinkle topping onto scones. bake
  15--20 minutes at \tempF[190--200]{375}.

  \begin{note}
    Use more milk for muffins.
  \end{note}
\end{recipe}
