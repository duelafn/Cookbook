
\begin{recipe}{Crepes}\tag{pancakes}
  \rating    4
  \nutrition 2
  \health    2
  \workload{medium}
  \time{60}
  \yield{6 crepes}
  \servings{2--3}
  \source{\Joy}
  % \photo{}
  \maketitle

  \begin{ingredients2}
    1 \cup & flour\\
    1 \cup & milk\\
    \half \cup & water\\
    4 & eggs\\
    \fourth \cup & butter\\
    \half \t & salt
  \end{ingredients2}

  Wisk all ingredients together. Let stand for 30 minutes. Fry in skillet.

  May fill with fruit sauce or nutella for breakfast or dessert. Fill with
  steak, feta cheese, summer squash, zucchini, and baby portabello mushrooms.
\end{recipe}
