
\begin{recipe}{Box Cake}\tag{cake}\tag{german chocolate cake}\tag{cake|german chocolate}
  \rating    4
  \nutrition 1
  \health    2
  \workload{light}
  \time{30}
  \yield{9x13 pan}
  \servings{12}
  \source{Dean}
  % \photo{}
  \maketitle

  \begin{ingredients2}
    1 & box cake mix (\qty{15.25}{oz})\\
    & box ingredients\\
    2 \T & cocoa powder (if chocolate)\\
    1\half \T & vanilla\\
    +1 & egg yolk\\
    $-2$ \T & water\\
  \end{ingredients2}

  Follow directions on box with added seasoning and one \textsl{extra} egg yolk. The
  Pillsbury devil's food cake works well for a chocolate cake with Betty Crocker
  ``Dark Chocolate'' frosting.
\end{recipe}
