
\begin{recipe}{Wheat Bread}\tag{bread}
  % \tag{}
  \rating    5
  \nutrition 3
  \health    4
  \workload{medium}
  \time{120}
  \source{\Joy}
  % \photo{}
  \maketitle

  \begin{ingredients2}
    \half \cup & warm water (no chlorine)\\
    2\fourth \t & yeast\\
    1 & egg\\
    2 \T & butter\\
    1 \cup & warm water\\
    \half \t & salt\\
    \fourth \cup & sugar/honey/syrup\\
    2 \cups & wheat flour\\
    2 \cups & white flour
  \end{ingredients2}

  knead, rise, split, trifold twice, shape, rise (45 min), bake 350{\deg}
  (205\deg--210\deg) in loaf. Cool 30 min.

\end{recipe}
