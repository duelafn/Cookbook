
\begin{recipe}{Chicken Soup}
  % \tag{}
  \rating    4
  \nutrition 4
  \health    4
  \workload{light}
  \time{10}
  \source{Alton Brown}
  % \photo{}
  \maketitle

  \begin{ingredients2}[3]
    chicken stock (p\pageref{Chicken Stock})\\
    onion\\
    celery\\
    garlic\\
    egg noodles
  \end{ingredients2}

  Simmer stock for 2 minutes. Add seasoning, simmer 5 minutes more. Turn
  off heat then add dry noodles.
\end{recipe}
