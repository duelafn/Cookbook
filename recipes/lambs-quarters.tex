
\begin{recipe}{Lambs Quarters}%\tag{}
  \rating    0
  \nutrition 3
  \health    3
  \workload{medium}
  \time{30}
  \yield{4~cups}
  \servings{4}
  \source{Sharda Brody}
  % \photo{}
  \maketitle

  \begin{ingredients2}
    4 \cups      & lambs quarters leaves\\
    1            & small or medium onion\\
    4            & bacon strips\\
    \fourth \cup & apple cider vinegar
 \end{ingredients2}

 Chop onion and bacon and fry until crisp. Pour off the grease. Combine
 with apple cider vinegar and simmer. Put greens in the sauce and ``wilt''
 the leaves.
\end{recipe}
