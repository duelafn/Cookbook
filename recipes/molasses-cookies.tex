
\begin{recipe}{Molasses Cookies}\tag{cookies}
  \rating    4
  \nutrition 1
  \health    2
  \workload{medium}
  \time{60}
  \yield{2 dozen 62\,g cookies}
  \servings{12}
  \source{\Joy}
  \photo{molasses-cookies}
  \maketitle

  \begin{ingredients2}
    1\half \cups & butter\\
    2 \cups & brown sugar\\
    4 \cups & all-purpose flour\\
    \half \t & salt\\
    4 \t & ground cinnamon\\
    2 \t & ground cloves\\
    2 \t & ground ginger\\
    2 \t & baking soda\\
    \half \cup & molasses\\
    2 & eggs\\
    & sugar for rolling
  \end{ingredients2}

  Beat butter and sugar, then add in rest of ingredients. Refrigerate until
  firm (2 hours). Roll into balls and roll in sugar. Bake on cookie sheet
  for approx 10 minutes at \tempF[190]{350}.
\end{recipe}
