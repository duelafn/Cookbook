
\begin{recipe}{Guacamole}%\tag{}
  \rating    0
  \nutrition 2
  \health    1
  \workload{medium}
  \time{60}
  \yield{}
  \servings{}
  \source{}
  % \photo{}
  \maketitle

  \begin{ingredients2}
    6 & avocado\\
    1 & lime\\
    \fourth \t & salt\\
    \fourth \t & cumin\\
    1 & chipotle\\
    2 & tomato\\
    2 clove & garlic\\
    1 bunch & cilantro
  \end{ingredients2}

  Chop avocado, juice lime on avocado, drain juice.

  Mash avocado with lime zest, 2~\T lime juice, and rest of ingredients.

  Let rest at room temperature for 1~hour.
\end{recipe}
