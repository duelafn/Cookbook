
\begin{recipe}{Pecan Pie}\tag{pie}
  \rating    4
  \nutrition 1
  \health    2
  \workload{medium}
  \time{90}
  \yield{1 pie}
  \servings{8}
  \source{Bonnie Schermer}
  % \photo{}
  \maketitle

  \begin{ingredients2}
    \twothird \cup & sugar\\
    \third \cup    & butter (melted)\\
    1 \cup         & corn syrup\\
    \half \t       & salt\\
    3              & eggs\\
    1 \cup         & pecan halves
  \end{ingredients2}

  Prepare a pie crust. Beat everything except pecans. Pour into pie shell.
  Arrange pecans on top. Bake 40 to 50 minutes at 375\degF until set.
\end{recipe}
