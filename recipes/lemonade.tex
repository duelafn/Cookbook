
\begin{recipe}{Lemonade}%\tag{}
  \rating    3
  \nutrition 0
  \health    0
  \workload{light}
  \time{5}
  \yield{10~cups}
  \servings{8}
  \source{}
  \maketitle
  % \photo{}

  \begin{ingredients}
    8 & lemons (1\half \cups concentrate)\\
    & zest of lemons\\
    1\half \cup & sugar\\
    & water to 1 gallon
  \end{ingredients}

  Heat 2~cups water to disolve sugar. Mix all in a pitcher. Prefer using lemons and
  avoid concentrate.

  \textsl{For a single glass}: 1 lemon, half its zest, 3~\T sugar, water to 16~\oz.
\end{recipe}
