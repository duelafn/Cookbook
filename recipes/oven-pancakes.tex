
\begin{recipe}{Oven Pancakes}\tag{pancakes}
  \rating    4
  \nutrition 2
  \health    2
  \workload{medium}
  \time{45}
  \yield{1 casserole}
  \servings{4}
  \source{Schermer}
  % \photo{}
  \maketitle

  \section{Pancake}
  \begin{ingredients2}
    3 \T & butter\\
    4 & eggs\\
    \twothird \cup & flour\\
    \fourth \t & salt\\
    \twothird \cup & milk
  \end{ingredients2}

  Preheat oven to 400\degF. Melt butter. Beat together with eggs, flour,
  salt, and milk. Apply oil to 10 inch casserole dish or iron skillet. Pour
  batter in pan and bake for 15 minutes.

  \section{Lemon Sauce}
  \begin{ingredients2}
    1 \T & butter\\
    1 \T & lemon juice\\
    1 \T & honey\\
    1 \cup & chopped nuts (optional)\\
    & powdered sugar
  \end{ingredients2}

  Melt butter and mix in lemon juice and honey. Pour mixture over pancake
  and sprinkle with powdered sugar. Cut and serve slices.
\end{recipe}
