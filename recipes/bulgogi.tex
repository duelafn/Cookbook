
\begin{recipe}{Bulgogi}\tag{beef}
  \rating    4
  \nutrition 3
  \health    3
  \workload{light}
  \time{30}
  \yield{}
  \servings{4}
  \source{Sunny's}
  \maketitle
  \photo{bulgogi}

  \begin{ingredients2}
    1\half \cup & rice\\
    2 \cups & water and chicken stock\\
    1 \lb & beef\\
    \half \cup & teriyaki sauce\\
    1 clove & garlic\\
    1 \t & minced ginger\\
    & green onion\\
    & sesame seeds\\
    & lettuce
  \end{ingredients2}

  Marinate thinly sliced beef in teriyaki for 2--3 hours.

  Over medium heat, cook garlig and ginger until fragrant. Spread beef mixture into
  wok evenly and cook for about 2 minutes per side. Serve with onions, sesame seeds,
  rice, and lettuce.
\end{recipe}
