
\begin{recipe}{Stuffing}
  \rating    4
  \nutrition 3
  \health    3
  \workload{light}
  \time{150}
  \yield{2 quarts}
  \servings{8}
  \source{Dean Serenevy}
  % \photo{}
  \maketitle

  \begin{ingredients2}
    18 \cups      & bread chunks (mixed flavors)\\
    \half \cup    & butter\\
    2             & eggs\\
    1\fourth \cup & chicken stock\\
    \half \lb     & mushrooms\\
    1 \lb         & sausage (cooked)\\
    \fourth bunch & fresh parsley\\
    several       & stalks celery\\
    sage, salt, pepper, rosemary\dots
  \end{ingredients2}

  Toss in oven-safe bowl. Bake \tempF[165]{350}. Approx 30--60 min, depends
  on initial temperature of liquids and shape/number of casseroles.

\end{recipe}
