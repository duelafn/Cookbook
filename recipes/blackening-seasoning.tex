
\begin{recipe}{Blackening Seasoning}
  % \tag{}
  \rating    5
  \workload{light}
  \time{10}
  \source{Dean Serenevy}
  % \photo{}
  \maketitle

  \begin{ingredients2}
    2 \t & cumin\\
    1 \t & salt\\
    1 \t & garlic powder\\
    1 \t & cayenne\\
    2 \t & black pepper\\
    \half \t & oregano\\
    \fourth \t & fennel
  \end{ingredients2}

  Mix. When cooking (grill or fry) use liberal amounts of butter.

\end{recipe}
