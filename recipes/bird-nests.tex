
\begin{recipe}{Bird Nests}\tag{cookies}
  \rating    4
  \nutrition 1
  \health    1
  \workload{medium}
  \time{60}
  \yield{}
  \servings{}
  \source{Bonnie Schermer}
  % \photo{}
  \maketitle

  \begin{ingredients2}
    12 \oz & butterscotch chips\\
    1 \cup & peanut butter\\
    15 \oz & chow mein noodles
  \end{ingredients2}

  In a dutch oven or double boiler, melt butterscotch chips. Add peanut
  butter, mix, and remove from heat. Add chow mein noodles. Spoon dollops
  onto wax paper and allow to cool.
\end{recipe}
