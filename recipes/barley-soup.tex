
\begin{recipe}{Barley Soup}%\tag{}
  \rating    5
  \nutrition
  \health
  \workload{light}
  \time{}
  \yield{}
  \servings{}
  \source{}
  \maketitle
  % \photo{}

  \begin{ingredients2}
    1--2 pkg & turkey sausage\\
    1 \cup & barley\\
    2 \cups & water\\
    3 \cups & chicken stock\\
    \half \t & salt\\
    2 & bay leaves\\
    \half \t & minced garlic\\
    1 \t & black pepper\\
    & frozen veggies
  \end{ingredients2}

  Fry up the sausage over medium-high heat. Deglaze pan, simmer everything
  40--50 minutes.
