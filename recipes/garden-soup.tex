
\begin{recipe}{Garden Soup}\tag{vegetarian}\tag{soup}
  \rating    3
  \nutrition 4
  \health    4
  \workload{light}
  \time{45}
  \yield{}
  \servings{6}
  \source{Hello Fresh}
  \photo{garden-soup}
  \maketitle

  \begin{ingredients2}
    12 oz & carrots\\
    2 cloves & garlic\\
    & italian seasonings\\
    28 oz & crushed tomato\\
    8 cups & water / stock\\
    4 oz & kale\\
    8 oz & couscous\\
    8 oz & grape tomatoes\\
    & other veggies
  \end{ingredients2}

  Caramelize carrots in some oil, add garlic, italian seasonings and any other
  long-cooking veggies. Cook briefly then add crushed tomatoes, and water~/~stock.
  Bring to boil, then simmer until carrots and veggies cooked. Add kale, couscous,
  tomatoes and any short-cooking veggies. Simmer until kale wilted and couscous done.
\end{recipe}
