
\begin{recipe}{Thai Shrimp Salad}\tag{seafood}
  \rating    3
  \nutrition 4
  \health    4
  \workload{medium}
  \time{60}
  \yield{}
  \servings{4}
  \source{Hello Fresh}
  \maketitle
  \photo{thai-shrimp-salad}

  \begin{ingredients2}
    12 \oz & carrots\\
    2 & baby cucumbers\\
    8 \oz & shredded red cabbage\\
    4 \oz & arugula\\
    1 & lime\\
    & sesame dressing\\
    24 \oz & shrimp\\
    4 cloves & garlic\\
    2 \oz & sweet Thai chili sauce
  \end{ingredients2}

  Shred carrots with grater and chop veggies, toss the salad with dressing
  and squeeze of lime.

  Cook shrimp with garlic until cooked. Add chili sauce, and half lime
  zest. Stir until coated and slightly carmelized.
\end{recipe}
