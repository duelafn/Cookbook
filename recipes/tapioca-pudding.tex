
\begin{recipe}{Tapioca Pudding}%\tag{}
  \rating    0
  \nutrition 1
  \health    2
  \workload{medium}
  \time{150}
  \yield{}
  \servings{}
  \source{}
  % \photo{}
  \maketitle

  \begin{ingredients2}
    1 \c & tapioca\\
    5 \c & whole milk\\
    1 \c & heavy cream\\
    2 & egg yolks\\
    \twothird \c & sugar\\
    & lemon zest\\
    2 \T & vanilla
  \end{ingredients2}

  Untested

  Combine tapioca, milk, and cream in crock pot and cook on high for 2 hours.

  Combine eggs, sugar, and lemon zest. \textsl{Temper with hot pudding}, then add to
  crock pot. Cook at 210\degF another 15 minutes. Add vanilla, stir, cover tightly
  with plastic wrap and cool on counter 1 hour, then move to refrigerator.
\end{recipe}
