
\begin{recipe}{Cauliflower Wraps}\tag{vegetarian}
  \rating    5
  \nutrition 4
  \health    3
  \workload{medium}
  \time{40}
  \yield{6 wraps}
  \servings{6}
  \source{Hello Fresh}
  \maketitle
  \photo{cauliflower-wraps}

  \begin{ingredients2}
    3 \lb & cauliflower (2 heads)\\
    2 \t & cumin\\
    2 \t & paprika\\
    1 \lb & red cabbage (\half head)\\
    1 \c & mayonnaise\\
    2 cloves & garlic\\
    1 \T & lemon juice\\
    1\half \T & honey\\
    1 \T & curry powder\\
    & tortillas\\
    & veggies or potatoes
  \end{ingredients2}

  Toss cauliflower with oil, cumin, and paprika. Roast in 450\degF oven for 20--25
  minutes. Combine mayo, garlic, lemon, honey, and curry. Toss with shredded cabbage.
  Wrap in tortillas and optionally bake 5 minutes to toast the tortillas. Serve with
  veggies or potatoes.
\end{recipe}
