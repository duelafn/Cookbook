
\begin{recipe}{Chocolate Muffins}\tag{muffins}
  \rating    4
  \nutrition 3
  \health    3
  \workload{medium}
  \time{45}
  \yield{24 muffins}
  \servings{24}
  \source{Schermer}
  % \photo{}
  \maketitle

  \begin{ingredients2}
    3 \cups     & flour\\
    2 \cups     & sugar\\
    6 \T        & unsweetened cocoa\\
    2 \t        & baking soda\\
    \half \cup  & oil\\
    2 \T        & vinegar (white or cider)\\
    2 \t        & vanilla\\
    2 \cups     & cold water
  \end{ingredients2}

  Pre-heat oven to 375\degF. Mix ingredients and apply oil to muffin tin.
  Pour in batter and bake 18 to 20 minutes. Batter can also be baked as a
  cake instead of as muffins.
\end{recipe}
