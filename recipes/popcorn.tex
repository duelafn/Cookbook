
\begin{recipe}{Popcorn}
  \rating    5
  \nutrition 3
  \health    2
  \workload{light}
  \time{15}
  \yield{10~cups}
  \servings{2}
  \source{}
  % \photo{}
  \maketitle

  \begin{ingredients2}
    \half \cup & poppcorn\\
    3 \T & butter (melted)\\
    1.5 pinch & salt
  \end{ingredients2}

  If using an air popper, start pouring butter while still popping for even
  coverage.
\end{recipe}
