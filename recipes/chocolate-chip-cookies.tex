
\begin{recipe}{Chocolate Chip Cookies}\tag{cookies}
  % \tag{}
  \rating    5
  \nutrition 2
  \health    2
  \workload{medium}
  \time{30}
  \yield{2 dozen 51\,g cookies}
  \source{Dean}
  \photo{chocolate-chip-cookies}
  \maketitle

  This is a soft and chewy remix of the standard recipe.

  \begin{ingredients2}
    1 \cup & softened butter\\
    \fourth \cup & sugar\\
    1\fourth \cup & brown sugar\\
    2 \t & vanilla\\
    2 & eggs\\
    2\fourth \cups & bread flour\\
    1 \t & baking soda\\
    12 \oz & chocolate chips
  \end{ingredients2}

  Cream butter, sugars, and vanilla. Add dry ingredients. Mix in mixer a bit longer
  than usual for gluten formation. Stir in chocolate chips, \emph{chill 1 hour},
  drop (do not smash down), bake 10-12 minutes at \tempF[190]{375}.

  \begin{history}
    % Exact recipe:
    % - 2 1/4 c  all-purpose flour
    % - 1 tsp    baking soda
    % - 1 tsp    salt
    % - 1 c      butter (softened)
    % - 3/4 c    sugar
    % - 3/4 c    brown sugar
    % - 1 tsp    vanilla
    % - 2 large  eggs
    % - 12 oz    chocolate chips
    The original toll-house recipe calls for all-purpose flour, \threefourth~\cup
    sugar, \threefourth~\cup brown sugar, and 1~\t vanilla.
  \end{history}
\end{recipe}
