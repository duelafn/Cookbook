
\begin{recipe}{Chocolate Chip Cookies}\tag{cookies}
  % \tag{}
  \rating    5
  \nutrition 2
  \health    2
  \workload{medium}
  \time{30}
  \yield{2 dozen 51\,g cookies}
  \source{Dean}
  % \photo{}
  \maketitle

  This is the soft and chewy remix of the standard recipe.

  \begin{ingredients2}
    1 \cup & \emph{melted} butter\\
    \fourth \cup & sugar\\
    1\fourth \cup & brown sugar\\
    1 \t & vanilla\\
    1 & egg\\
    1 & egg yolk\\
    2 \T & milk\\
    2\fourth \cups & bread flour\\
    1 \t & baking soda\\
    1 \t & salt
  \end{ingredients2}

  Mix in mixer a bit longer than usual (gluten formation), drop,
  \emph{chill 1 hour}, bake 9 minutes at \tempF[190]{375}.

  TO TRY: 1~c br sugar, 1/4~c sugar, 1/4~c corn syrup

  \begin{history}
    % Exact recipe:
    % - 2 1/4 c  all-purpose flour
    % - 1 tsp    baking soda
    % - 1 tsp    salt
    % - 1 c      butter (softened)
    % - 3/4 c    sugar
    % - 3/4 c    brown sugar
    % - 1 tsp    vanilla
    % - 2 large  eggs
    % - 12 oz    chocolate chips
    The original toll-house recipe calls for all-purpose flour, \threefourth~\cup
    sugar, \threefourth~\cup brown sugar, and 1~\t vanilla.
  \end{history}
\end{recipe}
